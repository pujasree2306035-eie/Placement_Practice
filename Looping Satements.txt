1) PRINTING NUMBERS FROM 1 TO N USING WHILE LOOP (POST INCREMENT):

#include<stdio.h>
int main()
{
    int n,itr;
    scanf("%d",&n);
    itr=1;
    while(itr<=n)
    {
        printf("%d\n",itr++);
    }
return 0;
}

11) PERFECT / ABUNDANT / DEFICIENT NUMBER:

#include<stdio.h>
int main()
{
    int i,n,sum=1;
    scanf("%d",&n);
    for(i=2;i<=n/2;i+=1)
    {

        if(n%i==0)
           sum+=i;
    }
    if(n==sum)
        printf("perfect");
    else if(sum>n)
        printf("abundant");
    else
        printf("deficient");
    return 0;
}
OUTPUT:
6
perfect

13) AMICABLE NUMBERS:

#include <stdio.h>
int main() {
    int n1, n2, sum1 = 0, sum2 = 0, i;
    scanf("%d %d", &n1, &n2);
    for (i = 1; i <= n1 / 2; i++) {
        if (n1 % i == 0)
            sum1 += i;
    }
    for (i = 1; i <= n2 / 2; i++) {
        if (n2 % i == 0)
            sum2 += i;
    }
    if (sum1 == n2 && sum2 == n1) {
        printf("Amicable Pair");
    } else {
        printf("Not Amicable Pair");
    }
    return 0;
}
OUTPUT:
3 4
Not Amicable Pair

16) STRONG NUMBER OR NOT:

#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int temp = n;   // store original number
    int fact, sum = 0;
    while (n != 0) {
        int dig = n % 10;
        fact = 1;
        for (int i = 1; i <= dig; i++) {
            fact *= i;   // factorial of digit
        }
        sum += fact;
        n /= 10;
    }
    if (sum == temp)   // compare with original number
        printf("strong");
    else
        printf("not");
    return 0;
}
OUTPUT:
145
strong
