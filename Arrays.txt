6.ARRAY TRAVERSAL LEFT TO RIGHT AND RIGHT TO LEFT

#include <stdio.h>
int main()
{
    int size;
    scanf("%d", &size);
    int arr[size];
    for (int i = 0; i < size; i++)
        scanf("%d", &arr[i]);
    // Left to Right
    for (int i = 0; i < size; i++)
        printf("%d ", arr[i]);
    printf("\n");
    // Right to Left
    for (int i = size - 1; i >= 0; i--)
        printf("%d ", arr[i]);
    return 0;
}

7.REVERSAL OF AN ARRAY:

#include <stdio.h>
int main()
{
    int size, start, end;
    scanf("%d", &size);
    int arr[size];
    for (int i = 0; i < size; i++)
        scanf("%d", &arr[i]);
    start = 0;
    end = size - 1;
    while (start < end)
    {
        int temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }
    for (int i = 0; i < size; i++)
        printf("%d ", arr[i]);
    return 0;
}
OUTPUT:
8
2 4 6 8 10 12 14 16
16 14 12 10 8 6 4 2

22.CHECK IDENTITY MATRIX:

#include <stdio.h>
int main()
{
    int n;
    scanf("%d", &n);  
    int A[n][n];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%d", &A[i][j]);
    int i, j;
    for (i = 0; i < n; i++) 
        {
        for (j = 0; j < n; j++) 
        {
            if (i == j)
            {
                if (A[i][j] != 1)   
                    break;
            } 
            else 
            {
                if (A[i][j] != 0)   
                    break;
            }
        }
        if (j < n)   
            break;
    }
    if (i < n)
        printf("Not Identity Matrix\n");
    else
        printf("Identity Matrix\n");
    return 0;
}

23.CHECK TOEPLITZ MATRIX:

#include <stdio.h>
int main() {
    int row, col;
    scanf("%d%d", &row, &col);
    int A[row][col];
    for (int i=0;i<row;i++)
        for (int j=0;j<col;j++)
            scanf("%d", &A[i][j]);
    int i, j;
    for (i=1;i<row;i++)
    {
        for (j=1;j<col-2;j++)
        {
            if (A[i][j] != A[i+1][j+1])
            {
                printf("Not a Toeplitz Matrix");
                return 0;
            }
        }
    }
    printf("Toeplitz Matrix");
    return 0;
}
OUTPUT:
3 3
1 2 3
3 1 2
2 3 1
Toeplitz Matrix


24.OUTER BOUNDARY AND DIAGONAL SUMS OF A MATRIX:

#include <stdio.h>
int main()
{
    int row,col,i,j;
    scanf("%d%d", &row,&col);
    int A[row][col];
    //input matrix
    for (int i = 0; i < row; i++)
        {
        for (int j = 0; j < col; j++)
            scanf("%d", &A[i][j]);
        }
        //process
        int outer = 0, left = 0, right = 0;
        for(int i=0;i<row;i++)
        {
            for(int j=0;j<col;j++)
            {
                 if (i == 0|| i == row - 1 || j==0 || j==col-1)
                    outer += A[i][j];
                 if (i ==j )
                    left += A[i][j];
                 else if (i+j == row-1)
                    right += A[i][j];
            }
        }
    printf("OUTER SUM: %d\n", outer);
    printf("LEFT DIAGONAL SUM: %d\n", left);
    printf("RIGHT DIAGONAL SUM: %d\n", right);
    return 0;
}
OUTPUT:
4 4
1 2 3 4
1 2 2 2
4 3 2 1
2 2 1 1
OUTER SUM: 24
LEFT DIAGONAL SUM: 6
RIGHT DIAGONAL SUM: 11
