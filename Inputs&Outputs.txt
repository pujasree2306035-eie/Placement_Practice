1. INTEGER INPUT:

#include<stdio.h>
int main()
{
    int a;
    scanf("%d",&a);
    printf("Integer value: %d",a);
    return 0;
}
OUTPUT:
6
Integer value: 6



2. FLOAT INPUT:

#include<stdio.h>
int main()
{
    float a;
    scanf("%f",&a);
    printf("Float value: %f",a);
    return 0;
}
OUTPUT:
34.77
Float value: 34.770000

5. NEW LINE ESCAPE SEQUENCE:

#include<stdio.h>
int main()
{
    int a,b;
    scanf("%d%d",&a,&b);
    printf("Two values are \n%d\n%d",a,b);
    return 0;
}
OUTPUT:
35 65
Two values are
35
65



6. TAB SPACE ESCAPE SEQUENCE:

#include<stdio.h>
int main()
{
    char a,b;
    scanf("%c %c",&a,&b);
    printf("Two characters\t%c\t%c",a,b);
    return 0;
}
OUTPUT:
A Z
Two characters  A       Z



7. CARRIAGE RETURN ESCAPE SEQUENCE:

#include<stdio.h>
int main()
{
    int a,b,c;
    scanf("%d%d%d",&a,&b,&c);
    printf("%d\r%d\r%d",a,b,c);
    return 0;
}
OUTPUT:
1000 530 47
4700
