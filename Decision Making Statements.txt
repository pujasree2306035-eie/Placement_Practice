DECISION MAKING PROGRAMS:

1. SIMPLE CALCULATOR USING IF–ELSE:

#include<stdio.h>
int main()
{
    int a,b;
    char ch;
    scanf("%d\n%d %c",&a,&b,&ch);
    if (ch == '+')
        printf("sum: %d",a+b);
    else if(ch == '-' )
        printf("difference: %d",a-b);
    else if(ch == '*')
        printf("product: %d",a*b);
     else if(ch == '/')
        printf("quotient: %d",a/b);
    else if(ch == '%')
        printf("remainder: %d",a%b);
    else
        printf("INVALID");
    return 0;
}
OUTPUT:
12
4
+
sum: 16


2. CHARACTER TYPE CHECKING:

#include<stdio.h>
int main()
{
    char ch;
    scanf("%c",&ch);
    if (ch>='A' && ch<='Z')
        printf("UPPERCASE");
    else if(ch>='a' && ch<='z')
        printf("lowercase");
    else if(ch>='0' && ch<='9')
        printf("Digit");
    else
        printf("Special characters");
    return 0;
}
OUTPUT:
a
lowercase


3. FINDING THE GREATEST OF THREE NUMBERS:

#include<stdio.h>
int main()
{
    int a,b,c;
    scanf("%d\n%d\n%d",&a,&b,&c);
    if (a>b && a>c)
        printf("a is greater");
    else if(b>a && b>c)
        printf("b is greater");
    else
        printf("c is greater");
    return 0;
}
OUTPUT:
34 56 78
c is greater


4. LEAP YEAR CHECKING USING NESTED IF:

#include<stdio.h>
int main()
{
    int year;
    scanf("%d",&year);
    if (year%4==0)
    {
        if (year%100==0)
        {
            if (year%400==0)
                printf("leap year");
            else
                printf("not a leap year");
        }
        else
            printf(" not leap year");
    }
    return 0;
}
OUTPUT:
2000
leap year

8. ARITHMETIC OPERATIONS USING IF–ELSE:

{
